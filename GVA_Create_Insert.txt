
CREATE TABLE Inventory ( ItemID int primary key, ItemDescription char(100), ItemCost int,InventoryDate char(30),ItemStorage Char(20),ItemLocation char(20));
CREATE TABLE Employee( EmployeeID int primary key, EmployeeFirstName char(30), EmployeeLastName char(30), EmployeePhone char(30));
CREATE TABLE Equipment( EquipmentID int primary key, EquipmentName char(30), EquipmentDescription char(100));
CREATE TABLE Supplies(SupplyID int primary key, SupplyType char(30));
CREATE Table Customer ( CustomerID int primary key, FirstName char(30), LastName char(30), CustomerAddress char(100), CustomerPhone char(30), PhoneType char(30) check(PhoneType in('Business','Cell','Home')),CustomerEmail char(30),HereAbout char(100),ContactThrough char(100), TotalEarnings int, DateAdded Datetime);
CREATE TABLE WorkFlow ( WorkFlowID int primary key, EngagmentDate datetime, CompletionDate date, Review char(500), CurrentStatus char(40) check(CurrentStatus in('Completed','In Progress', NULL)), CustomerID int references Customer not null, EmployeeID int references Employee);
CREATE TABLE ServiceEvent ( ServiceID int primary key,PotentialDate1 Date ,PotentialDate2 Date,ServiceDate Date,ServiceEndDate Date, ServiceCost int, ServiceType char(30) check(ServiceType in ('Moving','Auction')), WorkFlowID int references WorkFlow not null, ItemID int references Inventory, EquipmentID int references Equipment);
CREATE TABLE Auction (ServiceID int references ServiceEvent not null,Status char(30), SalePrice int, Address char(100),PRIMARY KEY(ServiceID));
CREATE TABLE Moving( ServiceID int references ServiceEvent not null, OriginAddress char(100), DestinationAddress char(100), PRIMARY KEY(ServiceID));
CREATE TABLE WorkFlowHistory(WFHistoryID int primary key,TicketSwitchDate date, TicketName char(30),TicketNote varchar(500) ,EmployeeID int references Employee,WorkFlowID int references WorkFlow);
CREATE TABLE CustomerRequest(CustomerRequestID int primary key, FirstName char(30), LastName Char(30), Email char(50),Street Char(30), City char(30), State char(30), Zip char(10),  Phone char(30), Status char(30) check(Status in('Completed','In Progress')));
CREATE TABLE ServiceRequest(ServiceRequestID int primary key, ServiceType char(30) check(ServiceType in ('Moving','Auction')), DateCreated DateTime,Description char(150), CustomerRequestID int references CustomerRequest );
Create Table InitialContact(ContactID int primary key, ReqServices varchar(50),LookAt char(10) check(LookAt in('Yes','No')), DownSize char(10) check(DownSize in('Yes','No')), Estate char(10) check(Estate in('Yes','No')), Moving char(10) check(Moving in('Yes','No')), QSold varchar(50), WhatSold varchar(50), ConversationNotes varchar(200), DateContacted datetime, CustomerID int references Customer);
Create Table MovingScreen(MovingScreenID int primary key, FinalMoveDate datetime, PricingSource varchar(20) check(PricingSource in('MLS','Look At','Phone','Email')), QuantityTrucks varchar(11), Trucks char(75), QuantityMovers varchar(11), Price char(20),Consumables char(20),EstimatedHours char(25),ActualHours char(25),OriginMiles char(25),DestMiles char(25), OtherMiles char(25),TotalMiles char(25),FuelCosts char(25), InsuranceCosts char(25),CostPerWorker char(25), Food char(25), WorkFlowID int references WorkFlow);
CREATE TABLE MSEmployee(MovingScreenID int references MovingScreen,EmployeeID int references Employee,primary key(MovingScreenID,EmployeeID));
Create Table AuctionPickUP(AuctionPickUpID int primary key, BringinDate datetime, PickupDate datetime, LookAtDate datetime, AppraisalDate datetime, BringInDateConfirmed datetime,PickUpDateConfirmed datetime, LookAtDateConfirmed datetime, AppraisalDateConfirmed datetime, SaleDate datetime, Location char(10),WorkFlowID int references WorkFlow);
CREATE TABLE CompletionForm (CompletionFormID	BIGINT IDENTITY(1,1) NOT NULL PRIMARY KEY,CustomerName VARCHAR(100) NOT NULL,SubTotal	VARCHAR(100) NOT NULL, AdditonalExpenses VARCHAR(100),Total VARCHAR(100) NOT NULL,PaidAmount VARCHAR(100) NOT NULL,Remaining VARCHAR(100),AdditionalNotes VARCHAR(500),EmployeeID INT references Employee(EmployeeID),EquipmentID INT references Equipment(EquipmentID),WorkFlowID INT references WorkFLow);
CREATE TABLE MoveForm (MoveFormID BIGINT IDENTITY(1,1) NOT NULL PRIMARY KEY, Consign VARCHAR(100), Smalls VARCHAR(100),  Stories int, SQF VARCHAR(100), BedroomQTY int, Bedrooms VARCHAR(100), LivingRoomQTY int, LivingRoom VARCHAR(100), DiningRoomQTY int, DiningRoom VARCHAR(100), OAttic VARCHAR(10), Attic VARCHAR(100), OBasement VARCHAR(10), Basement	VARCHAR(100), OGarage VARCHAR(10), Garage VARCHAR(100), OutBuilding VARCHAR(100), AwkwardItems VARCHAR(100), Collectibles VARCHAR(100), Appliances VARCHAR(100), Piano VARCHAR(10),LawnMower VARCHAR(10), DistanceDoor VARCHAR(100), Obstructions VARCHAR(100), AddNotes VARCHAR(100), WorkFlowID int references WorkFlow);
CREATE TABLE AuctionSchedulingForm(SchedulingFormID int primary key, LookAt char(10) check(LookAt in('Yes','No') ),Source char(10) check( Source in('Bring In','Pick Up') ), Distance char(50), DrieWay char(50),SmallBox int, MediumBox int, LargeBox int, ArtBox int, SmallPad int,LargePad int,NumberTrucks int,NumberMovers int,InventoryFile char(500),WorkFlowID int references WorkFlow);
CREATE TABLE AuctionDates(AuctionDateID int primary key, AuctionDate date);
CREATE TABLE ASSupplies(SchedulingFormID int references AuctionSchedulingForm,SupplyID int references Supplies,primary key(SchedulingFormID,SupplyID));
CREATE TABLE ASDate(SchedulingFormID int references AuctionSchedulingForm, AuctionDateID int references AuctionDates, primary key (SchedulingFormID,AuctionDateID));
CREATE TABLE ASEquipment(SchedulingFormID int references AuctionSchedulingForm,EquipmentID int references Equipment, primary key (SchedulingFormID,EquipmentID));
CREATE TABLE ASEmployee(SchedulingFormID int references AuctionSchedulingForm, EmployeeID  int references Employee,primary key (SchedulingFormID,EmployeeID ));
CREATE TABLE AuctionPhotos(PhotoID int Primary key, PhotoPath char(500),SchedulingFormID int references AuctionSchedulingForm);
CREATE TABLE CustomerNotes(CustomerNotesID BIGINT IDENTITY (1,1) Primary key, Notes varchar(500), CustomerID int references Customer);

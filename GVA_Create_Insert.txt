CREATE TABLE Inventory ( ItemID int primary key, ItemDescription char(100), ItemCost int,InventoryDate char(30));
CREATE TABLE Employee( EmployeeID int primary key, EmployeeFirstName char(30), EmployeeLastName char(30), EmployeePhone char(30));
CREATE TABLE Equipment( EquipmentID int primary key, EquipmentName char(30), EquipmentDescription char(100));
CREATE Table Customer ( CustomerID int primary key, FirstName char(30), LastName char(30), CustomerAddress char(100), CustomerPhone char(30), PhoneType char(30) check(PhoneType in('Business','Cell','Home')),CustomerEmail char(30),HereAbout char(100),ContactThrough char(100), TotalEarnings int, DateAdded Datetime); 
CREATE TABLE WorkFlow ( WorkFlowID int primary key, EngagmentDate datetime, CompletionDate date, Review char(500), CurrentStatus char(40), CustomerID int references Customer not null, EmployeeID int references Employee);
CREATE TABLE ServiceEvent ( ServiceID int primary key, ServiceDate Date,ServiceEndDate Date, ServiceCost int, ServiceType char(30) check(ServiceType in ('Moving','Auction')), WorkFlowID int references WorkFlow not null, ItemID int references Inventory, EquipmentID int references Equipment);
CREATE TABLE Auction (ServiceID int references ServiceEvent not null,Status char(30), SalePrice int, Address char(100),PRIMARY KEY(ServiceID));
CREATE TABLE Moving( ServiceID int references ServiceEvent not null, OriginAddress char(100), DestinationAddress char(100), PRIMARY KEY(ServiceID));
CREATE TABLE WorkFlowHistory(WFHistoryID int primary key,TicketSwitchDate date, TicketName char(30),TicketNote varchar(500) ,EmployeeID int references Employee,WorkFlowID int references WorkFlow);
CREATE TABLE CustomerRequest(CustomerRequestID int primary key, FirstName char(30), LastName Char(30), Email char(50),Street Char(30), City char(30), State char(30), Zip char(10),  Phone char(30), Status char(30) check(Status in('Completed','In Progress')));
CREATE TABLE ServiceRequest(ServiceRequestID int primary key, ServiceType char(30) check(ServiceType in ('Moving','Auction')), DateCreated DateTime,Description char(150), CustomerRequestID int references CustomerRequest );
